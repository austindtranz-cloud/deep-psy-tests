<script>
/* ═══════════════════════════════════════════
   DEEP PSY TESTS — ENGINE v10 (Universal Loader)
   Загружает все ресурсы с GitHub Pages.
   Автоматически определяет категорию по data-category-id.
   ═══════════════════════════════════════════ */

(function() {
  var baseUrl = "https://austindtranz-cloud.github.io/deep-psy-tests/";
  
  var resources = [
    { type: "style",  url: baseUrl + "core/style.css" },
    { type: "script", url: baseUrl + "data/test_registry.js" },
    { type: "script", url: baseUrl + "core/engine.js" },
    { type: "script", url: baseUrl + "core/engine_results.js" },
    { type: "script", url: baseUrl + "core/sidebar.js" },
    { type: "script", url: baseUrl + "core/success_modal.js" },
    { type: "script", url: baseUrl + "data/smil_566.js" },
    { type: "script", url: baseUrl + "data/leongard.js" }
  ];

  function loadResource(res) {
    if (res.type === "style") {
      var link = document.createElement("link");
      link.rel = "stylesheet";
      link.href = res.url;
      document.head.appendChild(link);
    } else {
      var script = document.createElement("script");
      script.src = res.url;
      script.async = false;
      document.head.appendChild(script);
    }
  }

  resources.forEach(loadResource);

  /* Universal auto-render for ALL category pages */
  document.addEventListener("DOMContentLoaded", function() {
    var checkReady = setInterval(function() {
      if (window.deepRenderCatalog && window.DEEP_MASTER_REGISTRY) {
        clearInterval(checkReady);

        /* Find container with data-category-id OR fallback to ID */
        var container = document.querySelector("[data-category-id]")
                     || document.getElementById("personality-categories-container");
        if (!container) return;

        var categoryId = container.getAttribute("data-category-id") || "personality";
        var containerId = container.id || "personality-categories-container";

        /* Auto-set DEEP_CATEGORY_DATA from master registry */
        if (window.DEEP_MASTER_REGISTRY[categoryId]) {
          window.DEEP_CATEGORY_DATA = window.DEEP_MASTER_REGISTRY[categoryId];
        }

        window.deepRenderCatalog(containerId);
      }
    }, 100);
  });
})();
</script>
