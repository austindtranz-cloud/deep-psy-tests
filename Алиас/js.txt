<script>
/* ═══════════════════════════════════════════
   DEEP PSY TESTS — ENGINE v11 (Smart Loader)
   Автоматическое обновление версии (Github API)
   ═══════════════════════════════════════════ */
(function() {
  var baseUrl = "https://austindtranz-cloud.github.io/deep-psy-tests/";

  function loadResources(v) {
    if (v) v = "?v=" + v;
    var resources = [
      { type: "style",  url: baseUrl + "core/style.css" + v },
      { type: "script", url: baseUrl + "data/test_registry.js" + v },
      { type: "script", url: baseUrl + "core/router.js" + v },
      { type: "script", url: baseUrl + "core/quiz_core.js" + v },
      { type: "script", url: baseUrl + "core/templates.js" + v },
      { type: "script", url: baseUrl + "core/nav_controller.js" + v },
      { type: "script", url: baseUrl + "core/grid_controller.js" + v },
      { type: "script", url: baseUrl + "core/ui_controller.js" + v },
      { type: "script", url: baseUrl + "core/integrations.js" + v },
      { type: "script", url: baseUrl + "core/engine_results.js" + v },
      { type: "script", url: baseUrl + "core/sidebar.js" + v },
      { type: "script", url: baseUrl + "core/success_modal.js" + v },
      { type: "script", url: baseUrl + "core/engine.js" + v }
    ];

    resources.forEach(function(res) {
      if (res.type === "style") {
        var link = document.createElement("link");
        link.rel = "stylesheet"; link.href = res.url;
        document.head.appendChild(link);
      } else {
        var script = document.createElement("script");
        script.src = res.url; script.async = false;
        document.head.appendChild(script);
      }
    });

    init();
  }

  var preloaderFailsafe;

  function injectFullpageLoader() {
    if (document.getElementById("deep-full-preloader")) return;
    
    var pl = document.createElement("div");
    pl.id = "deep-full-preloader";
    pl.innerHTML = '<style>' +
      '#deep-full-preloader { position:fixed; top:0; left:0; width:100%; height:100vh; background:#0f0f0f; z-index:2147483647; display:flex; flex-direction:column; align-items:center; justify-content:center; transition:opacity 0.6s ease; margin:0; padding:0; overflow:hidden; } ' +
      '.deep-glow { position:absolute; width:300px; height:300px; background:rgba(213,203,186,0.03); filter:blur(50px); border-radius:50%; animation:deep-pulse-glow 4s infinite alternate; } ' +
      '.deep-spinner-box { position:relative; z-index:2; display:flex; flex-direction:column; align-items:center; } ' +
      '.deep-spinner { width:40px; height:40px; border:2px solid rgba(213,203,186,0.1); border-top-color:rgba(213,203,186,0.8); border-radius:50%; animation:deep-spin 1s cubic-bezier(0.4, 0.1, 0.4, 1) infinite; margin-bottom:20px; } ' +
      '@keyframes deep-spin { to { transform:rotate(360deg); } } ' +
      '@keyframes deep-pulse-glow { from { transform:scale(0.9); opacity:0.6; } to { transform:scale(1.1); opacity:1; } } ' +
      '</style>' +
      '<div class="deep-glow"></div>' +
      '<div class="deep-spinner-box">' +
        '<div class="deep-spinner"></div>' +
        '<div style="color:rgba(213,203,186,0.4); font-family:system-ui, -apple-system, sans-serif; font-size:12px; letter-spacing:2px; text-transform:uppercase;">Загрузка</div>' +
      '</div>';
    
    var parent = document.body || document.documentElement;
    if (parent) parent.appendChild(pl);
    
    // FAILSAFE: Если через 4 секунды прелоадер всё ещё висит - убираем его принудительно
    preloaderFailsafe = setTimeout(removePreloader, 4000);
  }

  function removePreloader() {
    if (preloaderFailsafe) clearTimeout(preloaderFailsafe);
    var pl = document.getElementById("deep-full-preloader");
    if (pl) {
      pl.style.opacity = "0";
      setTimeout(function() { if (pl.parentNode) pl.parentNode.removeChild(pl); }, 600);
    }
  }

  function init() {
    var attempts = 0;
    var checkReady = setInterval(function() {
      attempts++;
      if (attempts > 100) { 
        clearInterval(checkReady); 
        removePreloader(); // FAILSAFE 2
        return; 
      }

      if (window.deepRenderCatalog && window.DEEP_MASTER_REGISTRY) {
        clearInterval(checkReady);
        var container = document.getElementById("deep-categories-container") || document.querySelector("[data-category-id]") || document.getElementById("personality-categories-container");
        if (!container) { 
          removePreloader();
          return; 
        }

        var categoryId = container.getAttribute("data-category-id");
        if (categoryId && window.DEEP_MASTER_REGISTRY[categoryId]) {
          window.DEEP_CATEGORY_DATA = window.DEEP_MASTER_REGISTRY[categoryId];
        }

        window.deepRenderCatalog(container.id);
        
        // Успешный рендер каталога — снимаем прелоадер!
        removePreloader();
      }
    }, 100);
  }

  // Смарт-лоадер: ищем закешированную версию сессии
  var CACHE_KEY = "deep_psy_version";
  var cachedV = null;
  try { cachedV = sessionStorage.getItem(CACHE_KEY); } catch(e) {}

  // Показываем прелоадер ТОЛЬКО при первом холодном запуске (когда нет кэша)
  if (!cachedV) {
    injectFullpageLoader();
  }

  // Исключаем лишние запросы к GitHub API, чтобы избежать лимитов (Rate Limiting).
  // Генерируем cache buster на основе текущего часа
  var currentHour = new Date().toISOString().slice(0, 13); // e.g., "2023-10-27T10"

  if (cachedV && cachedV.startsWith(currentHour)) {
    loadResources(cachedV);
  } else {
    var v = currentHour + "_" + Math.floor(Math.random() * 10000);
    try { sessionStorage.setItem(CACHE_KEY, v); } catch(e) {}
    loadResources(v);
  }
})();
</script>
