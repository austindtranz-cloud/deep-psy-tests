<script>
/* ═══════════════════════════════════════════
   DEEP PSY TESTS — ENGINE v11 (Smart Loader)
   Автоматическое обновление версии (Github API)
   ═══════════════════════════════════════════ */
(function() {
  var baseUrl = "https://austindtranz-cloud.github.io/deep-psy-tests/";
  
  /* HOTFIX: engine.js (старая версия на сервере) делает window.DEEP_TESTS = TESTS;
     где TESTS не объявлен → ReferenceError → deepRenderCatalog не создаётся.
     Объявляем TESTS в глобальной области ДО загрузки engine.js — ошибка исчезает. */
  window.TESTS = window.DEEP_TESTS || {};

  function loadResources(v) {
    if (v) v = "?v=" + v;
    var resources = [
      { type: "style",  url: baseUrl + "core/style.css" + v },
      { type: "script", url: baseUrl + "data/test_registry.js" + v },
      { type: "script", url: baseUrl + "core/engine.js" + v },
      { type: "script", url: baseUrl + "core/engine_results.js" + v },
      { type: "script", url: baseUrl + "core/sidebar.js" + v },
      { type: "script", url: baseUrl + "core/success_modal.js" + v },
      { type: "script", url: baseUrl + "data/smil_566.js" + v },
      { type: "script", url: baseUrl + "data/leongard.js" + v }
    ];

    resources.forEach(function(res) {
      if (res.type === "style") {
        var link = document.createElement("link");
        link.rel = "stylesheet"; link.href = res.url;
        document.head.appendChild(link);
      } else {
        var script = document.createElement("script");
        script.src = res.url; script.async = false;
        document.head.appendChild(script);
      }
    });

    init();
  }

  function init() {
    var attempts = 0;
    var checkReady = setInterval(function() {
      attempts++;
      if (attempts > 100) { clearInterval(checkReady); return; }

      if (window.deepRenderCatalog && window.DEEP_MASTER_REGISTRY) {
        clearInterval(checkReady);
        var container = document.querySelector("[data-category-id]") || document.getElementById("personality-categories-container");
        if (!container) return;

        var categoryId = container.getAttribute("data-category-id") || "personality";
        var containerId = container.id || "personality-categories-container";

        if (window.DEEP_MASTER_REGISTRY[categoryId]) {
          window.DEEP_CATEGORY_DATA = window.DEEP_MASTER_REGISTRY[categoryId];
        }

        window.deepRenderCatalog(containerId);
      }
    }, 100);
  }

  // Смарт-лоадер: ищем закешированную версию сессии, иначе спрашиваем GitHub
  var CACHE_KEY = "deep_psy_version";
  var cachedV = null;
  try { cachedV = sessionStorage.getItem(CACHE_KEY); } catch(e) {}

  if (cachedV) {
    loadResources(cachedV);
  } else {
    // Делаем микро-запрос к GitHub API (лимит 60/час с IP), чтобы узнать SHA последнего коммита.
    // Если на сайте много трафика, 60 уников с одного IP может упереться в лимит,
    // но Fallback сработает безопасно.
    fetch("https://api.github.com/repos/austindtranz-cloud/deep-psy-tests/commits/main")
      .then(function(r) { return r.json(); })
      .then(function(data) {
        var commitSha = (data && data.sha) ? data.sha.substring(0, 8) : "live";
        try { sessionStorage.setItem(CACHE_KEY, commitSha); } catch(e) {}
        loadResources(commitSha);
      })
      .catch(function() {
        // Ошибка сети или лимит Github API — грузим актуальное без кэшбастера (браузер сам решит)
        loadResources("");
      });
  }
})();
</script>
