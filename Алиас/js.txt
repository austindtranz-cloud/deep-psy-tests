<script>
/* ═══════════════════════════════════════════
   DEEP PSY TESTS — ENGINE v5 (Loader)
   Автоматическая загрузка движка с GitHub.
   Теперь включает: Поиск, Алфавит, CRM и новые тесты.
   ═══════════════════════════════════════════ */

(function() {
  var baseUrl = "https://austindtranz-cloud.github.io/deep-psy-tests/";
  
  // Список необходимых ресурсов
  var resources = [
    { type: "style",  url: baseUrl + "core/style.css" },
    { type: "script", url: baseUrl + "data/test_registry.js" },
    { type: "script", url: baseUrl + "core/engine.js" },
    { type: "script", url: baseUrl + "core/sidebar.js" },
    { type: "script", url: baseUrl + "data/smil_566.js" },
    { type: "script", url: baseUrl + "data/leongard.js" }
  ];

  function loadResource(res) {
    if (res.type === "style") {
      var link = document.createElement("link");
      link.rel = "stylesheet";
      link.href = res.url;
      document.head.appendChild(link);
    } else {
      var script = document.createElement("script");
      script.src = res.url;
      script.async = false; // Сохраняем порядок загрузки
      document.head.appendChild(script);
    }
  }

  // Загружаем всё по очереди
  resources.forEach(loadResource);

  // Инициализация при готовности DOM
  document.addEventListener("DOMContentLoaded", function() {
    var checkReady = setInterval(function() {
      if (window.deepRenderCatalog && window.DEEP_TEST_REGISTRY) {
        clearInterval(checkReady);
        // Если на странице есть контейнер для каталога — рендерим
        var container = document.getElementById("personality-categories-container");
        if (container) {
          window.deepRenderCatalog("personality-categories-container");
        }
      }
    }, 100);
  });
})();
</script>
